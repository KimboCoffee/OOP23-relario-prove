@startuml
package it.unibo.oop.relario {
    package model {

        

        package inventory {
            interface Inventory {
                + {method} useItem
                + {method} getItemName
                + {method} getItemDescription
                + {method} dropItem
                + {method} getItemsList
                + {method} addItem
                + {method} getItemsNumber
            }

            interface InventoryItem {
                + {method} getName
                + {method} getDescription
                + {method} getEffect
                + {method} isEquippable
                + {method} getDurability
            }

            interface Weapon extends InventoryItem {}

            interface Armor extends InventoryItem {}

            interface Healing extends InventoryItem {}
        }

        package entities {
            interface Entity {
                + {method} getPosition
            }

            interface NonPlayingEntity extends Entity {
                + {method} isInteractive
                + {method} interact
            }

            package living{
                interface PlayingCharacter extends Entity{
                    + {method} move
                    + {method} handleInventory
                    + {method} attacked
                    + {method} getLife
                }

                PlayingCharacter *-- Inventory

                interface NonPlayingCharacter extends NonPlayingEntity {
                    + {method} interact
                }

                interface Enemy extends NonPlayingCharacter {
                    + {method} dropsOnDeath
                    + {method} attacked
                    + {method} begMercy
                }
            }

            package furniture {
                interface FurnitureItem extends NonPlayingEntity {
                    + {method} isWalkable
                    + {method} isInteractive
                    + {method} interact
                }

                interface WalkableFurniture extends FurnitureItem {}

                interface ObstructingFurniture extends FurnitureItem {}
            }
        }
    }
}
@enduml
